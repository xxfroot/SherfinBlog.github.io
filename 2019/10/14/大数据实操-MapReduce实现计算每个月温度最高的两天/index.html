public class Main {

public static void main(String[] args) throws IOException, ClassNotFoundException, InterruptedException {

    Configuration configuration = new Configuration(true);

    Job job = Job.getInstance(configuration);

    job.setJarByClass(Main.class);

    job.setJobName("calWenDu");

    job.setMapperClass(MyMapper1.class);
    job.setInputFormatClass(TextInputFormat.class);
    job.setMapOutputKeyClass(Weather.class);
    job.setMapOutputValueClass(IntWritable.class);

    job.setPartitionerClass(MyPartitioner.class);
    job.setSortComparatorClass(MySort.class);

    job.setGroupingComparatorClass(MyGroup.class);
    job.setReducerClass(MyReducer1.class);

    job.setNumReduceTasks(2);

    job.setOutputKeyClass(Text.class);
    job.setOutputValueClass(IntWritable.class);

    Path input = new Path("/user/root/tem.txt");
    FileInputFormat.addInputPath(job,input);

    Path output = new Path("/user/root/tem_out.txt");
    FileOutputFormat.setOutputPath(job,output);
    if (output.getFileSystem(configuration).exists(output)){
        output.getFileSystem(configuration).delete(output,true);
    }
    job.waitForCompletion(true);
}

}

    this.month = month;

    }

    return day;

    public void setDay(int day) {

    this.day = day;

    }

    return temerapure;

    }

    public void setTemerapure(int temerapure) {

    this.temerapure = temerapure;

@Override
public int compareTo(Weather o) {
    int r = Integer.compare(this.year,o.getYear());
    if (r==0){
        int r1=Integer.compare(this.month,o.getMonth());
        if (r1==0){
            return Integer.compare(this.day,o.getDay());
        }
        return r1;
    }
    return r;
}

@Override
public void write(DataOutput out) throws IOException {
    out.writeInt(this.year);
    out.writeInt(this.month);
    out.writeInt(this.day);
    out.writeInt(this.temerapure);
}

@Override
public void readFields(DataInput in) throws IOException {
    this.year=in.readInt();
    this.month=in.readInt();
    this.day=in.readInt();
    this.temerapure=in.readInt();
}

}

public class MyMapper1 extends Mapper<LongWritable, Text,Weather, IntWritable> {

Weather weather = new Weather();

IntWritable intWritable = new IntWritable();

@Override
protected void map(LongWritable key, Text value, Context context) throws IOException, InterruptedException {
    String[] strs = value.toString().split("    ");

    SimpleDateFormat sdf = new SimpleDateFormat("yyyy-mm-dd");

    try {
        Date date =  sdf.parse(strs[0]);
        Calendar calendar = Calendar.getInstance();
        calendar.setTime(date);
        weather.setYear(calendar.get(Calendar.YEAR));
        weather.setMonth(calendar.get(Calendar.MONTH+1));
        weather.setDay(calendar.get(Calendar.DAY_OF_MONTH));

        int webdu=Integer.parseInt(strs[1]);
        weather.setTemerapure(webdu);

        intWritable.set(webdu);
        context.write(weather,intWritable);
    } catch (ParseException e) {
        e.printStackTrace();
    }

}

}

public class MySort extends WritableComparator {

public MySort(){
    super(Weather.class,true);
}

@Override
public int compare(WritableComparable a, WritableComparable b) {
    Weather weather1 = (Weather)a;
    Weather weather2 = (Weather)b;

    int r1=Integer.compare(weather1.getYear(),weather2.getYear());
    if (r1==0){
        int r2=Integer.compare(weather1.getMonth(),weather2.getMonth());
            if (r2==0){
                return -Integer.compare(weather1.getTemerapure(),weather2.getTemerapure());
            }
            return r2;
    }
    return r1;
}

}

public class MyReducer1 extends Reducer <Weather, IntWritable, Text,IntWritable>{

Text rkey = new Text();

IntWritable intWritable = new IntWritable();

@Override
protected void reduce(Weather key, Iterable<IntWritable> values, Context context) throws IOException, InterruptedException {

            int flag=0;
            int day=0;
            for (IntWritable writable : values){
                if (flag==0){
                    rkey.set(key.getYear()+"-"+key.getMonth()+"-"+key.getDay()+":"+key.getTemerapure());
                    intWritable.set(key.getTemerapure());
                    flag++;
                    day=key.getDay();
                    context.write(rkey,intWritable);
                }
                if (flag!=0 && day!=key.getDay()){
                    rkey.set(key.getYear()+"-"+key.getMonth()+"-"+key.getDay()+":"+key.getTemerapure());
                    intWritable.set(key.getTemerapure());
                    context.write(rkey,intWritable);
                    break;
                }
            }

}

}
